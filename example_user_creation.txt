Create Database for Code Redemption:

CREATE DATABASE `tt_code_redemption` /*!40100 COLLATE 'latin1_swedish_ci' */;

Create User for Code Redemption:

CREATE USER 'toontown_code_redemption'@'%' IDENTIFIED BY 'PASSWORD';

FLUSH PRIVILEGES;
